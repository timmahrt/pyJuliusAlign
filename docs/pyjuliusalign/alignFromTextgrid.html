 26def getSilencesInAudio(audioFN):

 29    silences = silence.detect_silence(

 36def invertIntervalList(inputList, minValue, maxValue):
 37    """
 38    Inverts the segments of a list of intervals
 39
 40    e.g.
 41    [(0,1), (4,5), (7,10)] -> [(1,4), (5,7)]
 42    [(0.5, 1.2), (3.4, 5.0)] -> [(0.0, 0.5), (1.2, 3.4)]
 43    """
 44    inputList = sorted(inputList)
 45
 46    # Special case -- empty lists
 47    if len(inputList) == 0 and minValue is not None and maxValue is not None:
 48        invList = [
 49            (minValue, maxValue),
 50        ]
 51    else:
 52        # Insert in a garbage head and tail value for the purpose
 53        # of inverting, in the range does not start and end at the
 54        # smallest and largest values
 55        if minValue is not None and inputList[0][0] > minValue:
 56            inputList.insert(0, (-1, minValue))
 57        if maxValue is not None and inputList[-1][1] < maxValue:
 58            inputList.append((maxValue, maxValue + 1))
 59
 60        invList = [
 61            (inputList[i][1], inputList[i + 1][0]) for i in range(0, len(inputList) - 1)
 62        ]
 63
 64        # If two intervals in the input share a boundary, we'll get invalid intervals in the output
 65        # eg invertIntervalList([(0, 1), (1, 2)]) -> [(1, 1)]
 66        invList = [interval for interval in invList if interval[0] != interval[1]]
 67
 68    return invList

153def bumpUltrashortSegments(speechSegments):
154    newSpeechSegments = []
155    minDuration = 0.30
156    for start, stop in speechSegments:
157        if stop - start < minDuration:
158            stop = start + minDuration
159        newSpeechSegments.append((start, stop))
160
161    # The silence should be large enough to accomodate the bump
162    # but validate this is the case
163    i = 0
164    while i < len(newSpeechSegments) - 1:
165        assert newSpeechSegments[i][1] < newSpeechSegments[i + 1][0]
166        i += 1
167
168    return newSpeechSegments

171def copyChunking(unchunkedList, modelChunking):
172    """
173    Chunk an unchunked list according to a chunked list
174
175    Bad things will probably happen if they don't have the same number of elements (when flattened)
176    """
177    copiedChunking = []
178    for modelChunk in modelChunking:
179        chunk = []
180        for i in range(len(modelChunk)):
181            chunk.append(unchunkedList.pop(0))
182
183        copiedChunking.append(chunk)
184
185    return copiedChunking

188def getBestChunking(
189    chunks,
190    bucketDurations,
191):
192    """
193    Given a list of chunks and a list of buckets, return the optimal placement of chunks into buckets
194
195    chunks and buckets are both linearly ordered; the order cannot be mixed
196    The sum of the chunks may be smaller or larger than the buckets
197    chunks -> a list of integers
198    bucketDurations -> a list of integers
199
200    This function returns a grouping of the chunks that is optimal for the buckets
201    (minimizes local overfilling or underfilling of buckets)
202    """
203    bestChunking = None
204    minScore = 100000
205    for chunking in iterateChunks(chunks, len(bucketDurations)):
206        chunkingSums = [sum(chunk) for chunk in chunking]
207        score = sum(
208            [
209                abs(bucketDuration - chunkDuration)
210                for bucketDuration, chunkDuration in zip(bucketDurations, chunkingSums)
211            ]
212        )
213        if score < minScore:
214            bestChunking = chunking
215            minScore = score
216
217    return bestChunking

220def iterateChunks(chunks, numBuckets):
221    """
222
223    This could probably be solved more efficiently
224    with a graph but for small chunks and numBuckets
225    it should be ok
226
227    The number of iterations per call is
228    numChunks - numBuckets + 1
229    (I don't have a good intution for why, but calculated it out on paper)
230    7 chunks; 4 buckets
231    [1, ...]
232    [1 2, ...]
233    [1 2 3, ...]
234    [1 2 3 4, 5, 6, 7]
235    7 - 4 + 1 -> 4 iterations
236    """
237    # print((chunks, numBuckets))
238    # exit(0)
239    gap = len(chunks) - numBuckets
240
241    if numBuckets == 1:
242        yield [chunks]  # [[4, 5, 6]]
243
244    elif len(chunks) == numBuckets:
245        yield [[chunk] for chunk in chunks]  # [[4], [5], [6]]
246
247    else:
248        for i in range(
249            1, gap + 2
250        ):  # + 2 b/c 1 for the 0-based offset; 1 for the algorithm (see doc string)
251            for tailChunk in iterateChunks(chunks[i:], numBuckets - 1):
252                yield [chunks[:i]] + tailChunk  # [[1, 2, 3]] + [[4], [5], [6]]

255def textgridToCSV(inputPath, outputPath, outputExt=".csv", tierName="utterances"):
256    utils.makeDir(outputPath)
257
258    for fn in utils.findFiles(inputPath, filterExt=".TextGrid"):
259        tg = textgrid.openTextgrid(join(inputPath, fn), includeEmptyIntervals=False)
260        tier = tg.getTier(tierName)
261        outputList = []
262        for start, stop, label in tier.entries:
263            outputList.append("%s,%s,%s" % (start, stop, label))
264
265        name = os.path.splitext(fn)[0]
266        outputTxt = "\n".join(outputList)
267        outputFN = join(outputPath, "%s%s" % (name, outputExt))
268        with io.open(outputFN, "w", encoding="utf-8") as fd:
269            fd.write(outputTxt)

356def forceAlignFile(
357    speakerList,
358    wavPath,
359    wavNameDict,
360    txtPath,
361    txtFN,
362    outputPath,
363    outputWavName,
364    juliusScriptPath,
365    soxPath,
366    perlPath,
367):
368    """
369
370    Normally:
371    speakerList = [name]
372    and
373    wavNameDict = {name:"name.wav"}
374
375    But, if you have multiple speakers for each file (assuming audio is synced)
376    e.g. in a stereo audio situation:
377    speakerList=["L","R"]
378    and
379    wavNameDict={"L":"%s_%s.wav" % (name, "L"), "R":"%s_%s.wav" % (name, "R")}
380    """
381
382    utils.makeDir(outputPath)
383
384    # Formatted output of cabocha
385    with io.open(join(txtPath, txtFN), "r", encoding="utf-8") as fd:
386        data = fd.read()
387    dataList = data.split("\n")
388    dataList = [
389        [subRow.split(",") for subRow in row.split(";")]
390        for row in dataList
391        if row != ""
392    ]
393
394    dataDict = {speaker: [] for speaker in speakerList}
395
396    # Undoing the unnecessary split that just happened
397    for timingInfo, line, wordList, kanaList, romajiList in dataList:
398        line = ",".join(line)
399
400        speaker, startTimeStr, endTimeStr = timingInfo
401        speaker, startTime, endTime = (
402            speaker.strip(),
403            float(startTimeStr),
404            float(endTimeStr),
405        )
406
407        dataDict[speaker].append(
408            [startTime, endTime, line, wordList, kanaList, romajiList]
409        )
410
411    # Do the forced alignment
412    speakerEntryDict = {}
413    numPhonesFailedAlignment = 0
414    numPhones = 0
415    numFailedIntervals = 0
416    numIntervals = 0
417    for speaker in speakerList:
418        output = juliusAlignment.juliusAlignCabocha(
419            dataDict[speaker],
420            wavPath,
421            wavNameDict[speaker],
422            juliusScriptPath,
423            soxPath,
424            perlPath,
425            False,
426            True,
427            True,
428        )
429        speakerEntryDict[speaker], statList = output
430
431        numPhonesFailedAlignment += statList[0]
432        numPhones += statList[1]
433        numFailedIntervals += statList[2]
434        numIntervals += statList[3]
435
436    # All durations should be the same
437    inputWavFN = next(iter(wavNameDict.values()))
438    maxDuration = audioScripts.getSoundFileDuration(join(wavPath, inputWavFN))
439
440    # Create tiers and textgrids from the output of the alignment
441    tg = textgrid.Textgrid()
442    for speaker in speakerList:
443        for aspect in [
444            juliusAlignment.UTTERANCE,
445            juliusAlignment.WORD,
446            juliusAlignment.PHONE,
447        ]:
448            tierName = "%s_%s" % (aspect, speaker)
449            tier = textgrid.IntervalTier(
450                tierName, speakerEntryDict[speaker][aspect], minT=0, maxT=maxDuration
451            )
452            tg.addTier(tier)
453
454    tg.save(
455        join(outputPath, outputWavName + ".TextGrid"),
456        format="short_textgrid",
457        includeBlankSpaces=True,
458    )
459
460    return (numPhonesFailedAlignment, numPhones, numFailedIntervals, numIntervals)

190    chunks,
191    bucketDurations,
192):
193    """
194    Given a list of chunks and a list of buckets, return the optimal placement of chunks into buckets
195
196    chunks and buckets are both linearly ordered; the order cannot be mixed
197    The sum of the chunks may be smaller or larger than the buckets
198    chunks -> a list of integers
199    bucketDurations -> a list of integers
200
201    This function returns a grouping of the chunks that is optimal for the buckets
202    (minimizes local overfilling or underfilling of buckets)
203    """
204    bestChunking = None
205    minScore = 100000
206    for chunking in iterateChunks(chunks, len(bucketDurations)):
207        chunkingSums = [sum(chunk) for chunk in chunking]
208        score = sum(
209            [
210                abs(bucketDuration - chunkDuration)
211                for bucketDuration, chunkDuration in zip(bucketDurations, chunkingSums)
212            ]
213        )
214        if score < minScore:
215            bestChunking = chunking
216            minScore = score
217
218    return bestChunking

226    it should be ok
227
228    The number of iterations per call is
229    numChunks - numBuckets + 1

231    7 chunks; 4 buckets
232    [1, ...]

234    [1 2 3, ...]

236    7 - 4 + 1 -> 4 iterations
237    """
238    # print((chunks, numBuckets))
239    # exit(0)
240    gap = len(chunks) - numBuckets
241
242    if numBuckets == 1:

244
245    elif len(chunks) == numBuckets:
